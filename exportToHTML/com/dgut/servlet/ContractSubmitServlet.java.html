11   import java.io.IOException;
12   import java.io.Serial;
13   import java.sql.SQLException;
14   import java.util.ArrayList;
15   import java.sql.Date;
16   import java.util.List;
17   
18   import javax.servlet.ServletException;
19   import javax.servlet.annotation.WebServlet;
20   import javax.servlet.http.HttpServlet;
21   import javax.servlet.http.HttpServletRequest;
22   import javax.servlet.http.HttpServletResponse;
23   
24   //@WebServlet("/contract/submit")
25   public class ContractSubmitServlet extends HttpServlet {
26   
27       @Serial
28       private static final long serialVersionUID = 1L;
29   
30       protected void doPost(HttpServletRequest request, HttpServletResponse response)
31               throws ServletException, IOException {
32           // 读取表单数据
33           String customerId = request.getParameter("customer_id");
34           String salespersonId = request.getParameter("salesperson_id");
35           Date startDate = Date.valueOf(request.getParameter("start_date"));
36           Date endDate = Date.valueOf(request.getParameter("end_date"));
37           String status = request.getParameter("status");
38   
39           // 创建合同对象
40           Contract contract = new Contract();
41           contract.setCustomerId(Integer.parseInt(customerId));
42           contract.setSalespersonId(Integer.parseInt(salespersonId));
43           contract.setStartDate(startDate);
44           contract.setEndDate(endDate);
45           contract.setStatus(status);
46   
47           // 获取商品信息
48           List<PurchaseListItem> purchaseListItems = new ArrayList<>();
49           // 获取所有被选中的商品
50           String[] selectedGoodsIndex = request.getParameterValues("selectedGoodsIndex");
51           String[] goodsCount = request.getParameterValues("goodsCount");
52           Double totalAmount = 0.0;
53           if (selectedGoodsIndex != null) {
54               // 循环遍历所有被选中的商品
55               for (String index : selectedGoodsIndex) {
56                   int i = Integer.parseInt(index);
57                   GoodsDaoImpl goodsDao = new GoodsDaoImpl();
58                   List<Goods> goodsList = goodsDao.findAll();
59                   Goods goods = goodsList.get(i);
60   //                int quantity = Integer.parseInt(request.getParameter("goodsCount" + i));
61                   int quantity = Integer.parseInt(goodsCount[i]);
62                   double price = goods.getPrice();
63                   double subtotal = price * quantity;
64                   totalAmount += subtotal;
65                   PurchaseListItem purchaseListItem = new PurchaseListItem();
66                   purchaseListItem.setGoods(goods);
67                   purchaseListItem.setGoodsId(goods.getId());
68                   purchaseListItem.setQuantity(quantity);
69                   purchaseListItem.setSubtotal(subtotal);
70                   purchaseListItems.add(purchaseListItem);
71               }
72   
73               // 创建合同
74   
75               contract.setAmount(totalAmount);
76   
77               // 创建采购清单
78               PurchaseList purchaseList = new PurchaseList();
79               contract.setPurchaseListId(purchaseList.getId());
80   //            purchaseList.setContract(contract);
81               purchaseList.setPurchaseListItems(purchaseListItems);
82   
83   
84               try {
85                   ContractDaoImpl contractDaoImpl = new ContractDaoImpl();
86                   contractDaoImpl.createContractAndPurchaseList(contract, purchaseList, purchaseListItems);
87               } catch (SQLException e) {
88                   throw new RuntimeException(e);
89               }
90   
91   // 重定向到合同列表页面
92               response.sendRedirect(request.getContextPath() + "/creatContract.jsp");
93   
94           }
95       }
96   }
