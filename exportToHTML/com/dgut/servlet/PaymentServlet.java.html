41       protected void doGet(HttpServletRequest request, HttpServletResponse response)
42               throws ServletException, IOException {
43           // Show payment form
44           int contractId = Integer.parseInt(request.getParameter("contract_id"));
45           Contract contract = contractDao.getContractById(contractId);
46           request.setAttribute("contract", contract);
47           request.getRequestDispatcher("/paymentForm.jsp").forward(request, response);
48       }
49   
50       protected void doPost(HttpServletRequest request, HttpServletResponse response)
51               throws ServletException, IOException {
52   
53           // 读取表单数据
54           int contractId = Integer.parseInt(request.getParameter("payment_contract_id"));
55           int purchaseListId = -1;
56   
57           // 获取商品信息
58           List<PurchaseListItem> purchaseListItems = new ArrayList<>();
59           // 获取所有被选中的商品
60           String[] selectedGoodsIndex = request.getParameterValues("selectedPaymentsIndex");
61           String[] goodsCount = request.getParameterValues("paymentsCount");
62           Double totalAmount = 0.0;
63           if (selectedGoodsIndex != null) {
64               // 循环遍历所有被选中的商品
65               for (String index : selectedGoodsIndex) {
66                   int i = Integer.parseInt(index);
67                   GoodsDaoImpl goodsDao = new GoodsDaoImpl();
68                   List<Goods> goodsList = goodsDao.findAll();
69                   Goods goods = goodsList.get(i);
70   //                int quantity = Integer.parseInt(request.getParameter("goodsCount" + i));
71                   int quantity = Integer.parseInt(goodsCount[i]);
72                   double price = goods.getPrice();
73                   double subtotal = price * quantity;
74                   totalAmount += subtotal;
75                   PurchaseListItem purchaseListItem = new PurchaseListItem();
76                   purchaseListItem.setGoods(goods);
77                   purchaseListItem.setGoodsId(goods.getId());
78                   purchaseListItem.setQuantity(quantity);
79                   purchaseListItem.setSubtotal(subtotal);
80                   purchaseListItems.add(purchaseListItem);
81               }
82   
83               try {
84                   purchaseListId = purchaseListDao.insertPurchaseListItem(purchaseListItems);
85               } catch (SQLException e) {
86                   throw new RuntimeException(e);
87               }
88   
89               // 创建合同
90               Payment payment = new Payment(contractId, purchaseListId, totalAmount, Date.valueOf(LocalDate.now()));
91               PaymentDao paymentDao = new PaymentDao();
92               try {
93                   paymentDao.insertPayment(payment);
94               } catch (SQLException ex) {
95                   throw new RuntimeException(ex);
96               }
97   
98   
99   // 重定向到合同列表页面
100              response.sendRedirect(request.getContextPath() + "/creatPayment.jsp");
101  
102          }
103      }
